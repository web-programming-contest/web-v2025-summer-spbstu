
// Полный калькулятор со скобками

S -> E$     // Start -> Expression eof

E -> TQ     // Expression -> Term NextTerm
 // Применено устранение левой рекурсии
 Q -> +TQ   // NextTerm -> addop Term NextTerm
 Q ->       // NextTerm ->

T -> FR     // Term -> Factor NextFactor
 // Применено устранение левой рекурсии
 R -> *FR   // NextFactor -> mulop Factor NextFactor
 R ->       // NextFactor ->

F -> P      // Factor -> Primary
F -> +F     // Factor -> addop Factor

P -> 1      // Primary -> integer
P -> (E)    // Primary -> '(' Expression ')'
